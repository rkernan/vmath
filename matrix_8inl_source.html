107 template<typename U, typename>
108 T Matrix<T, M, N>::minor(const std::size_t row, const std::size_t col) const {
109  Matrix<T, M - 1, N - 1> minor_mat;
110  std::size_t i = (col == 0) ? 1 : 0;
111  std::size_t i_minor = 0;
112  while (i_minor < N - 1) {
113  std::size_t j_minor = 0;
114  std::size_t j = (row == 0) ? 1 : 0;
115  while (j_minor < M - 1) {
116  minor_mat[i_minor][j_minor] = this->data[i][j];
117  j_minor++;
118  j = (j + 1 == row) ? (j + 2) : (j + 1);
119  }
120  i_minor++;
121  i = (i + 1 == col) ? (i + 2) : (i + 1);
122  }
123  return minor_mat.det();
124 }
